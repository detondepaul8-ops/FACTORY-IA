   224→// Simple markdown renderer
   225→function renderMd(text: string): string {
   226→  let html = text
   227→    .replace(/&/g, '&amp;')
   228→    .replace(/</g, '&lt;')
   229→    .replace(/>/g, '&gt;')
   230→
   231→  // Code blocks
   232→  html = html.replace(/```(\w*)\n([\s\S]*?)```/g, (_match, lang, code) => {
   233→    return `<div class="chat-code"><div class="flex items-center justify-between mb-2 px-1 text-xs text-factory-muted"><span>${lang || 'code'}</span></div><pre class="whitespace-pre-wrap text-sm text-factory-green/90">${code.trim()}</pre></div>`
   234→  })
   235→
   236→  // Inline code
   237→  html = html.replace(/`([^`]+)`/g, '<code class="bg-factory-card border border-factory-border px-1.5 py-0.5 rounded text-sm font-mono text-factory-green">$1</code>')
   238→
   239→  // Headings
   240→  html = html.replace(/^### (.+)$/gm, '<h3 class="text-lg font-semibold mt-4 mb-2 text-foreground">$1</h3>')
   241→  html = html.replace(/^## (.+)$/gm, '<h2 class="text-xl font-bold mt-4 mb-2 text-foreground">$1</h2>')
   242→  html = html.replace(/^# (.+)$/gm, '<h1 class="text-2xl font-bold mt-4 mb-2 text-foreground">$1</h1>')
   243→
   244→  // Bold and italic
   245→  html = html.replace(/\*\*(.+?)\*\*/g, '<strong class="font-semibold text-foreground">$1</strong>')
   246→  html = html.replace(/\*(.+?)\*/g, '<em>$1</em>')
   247→
   248→  // Lists
   249→  html = html.replace(/^[\-•] (.+)$/gm, '<li class="ml-4 list-disc text-muted-foreground">$1</li>')
   250→  html = html.replace(/^\d+\. (.+)$/gm, '<li class="ml-4 list-decimal text-muted-foreground">$1</li>')
   251→
   252→  // Paragraphs (lines that aren't already wrapped)
   253→  html = html.replace(/^(?!<[h|l|d|p|u])(.+)$/gm, '<p class="text-muted-foreground leading-relaxed">$1</p>')
   254→
   255→  // Clean up excessive newlines
   256→  html = html.replace(/<p class="text-muted-foreground leading-relaxed"><\/p>/g, '')
   257→
   258→  return html
   259→}

   461→function ChatTab({ user, onUpdateCredits }: { user: User; onUpdateCredits: (c: number) => void }) {
   462→  const [messages, setMessages] = useState<ChatMessage[]>([])
   463→  const [input, setInput] = useState('')
   464→  const [loading, setLoading] = useState(false)
   465→  const scrollRef = useRef<HTMLDivElement>(null)
   466→
   467→  useEffect(() => {
   468→    if (scrollRef.current) {
   469→      scrollRef.current.scrollTop = scrollRef.current.scrollHeight
   470→    }
   471→  }, [messages])
   472→
   473→  const sendMessage = async () => {
   474→    const trimmed = input.trim()
   475→    if (!trimmed || loading) return
   476→
   477→    const userMsg: ChatMessage = { id: uid(), role: 'user', content: trimmed }
   478→    const loadingMsg: ChatMessage = { id: uid(), role: 'assistant', content: '', loading: true }
   479→
   480→    setMessages((prev) => [...prev, userMsg, loadingMsg])
   481→    setInput('')
   482→    setLoading(true)
   483→
   484→    try {
   485→      const res = await apiFetch('/api/ask', {
   486→        method: 'POST',
   487→        body: JSON.stringify({ message: trimmed }),
   488→      })
   489→      const data = await res.json()
   490→
   491→      if (!res.ok) {
   492→        setMessages((prev) =>
   493→          prev.map((m) =>
   494→            m.id === loadingMsg.id
   495→              ? { ...m, loading: false, content: `❌ ${data.error || 'Erreur inconnue'}` }
   496→              : m
   497→          )
   498→        )
   499→        if (res.status === 402) toast.error('Crédits insuffisants ! Rechargez votre compte.')
   500→        return
   501→      }
   502→
   503→      setMessages((prev) =>
   504→        prev.map((m) =>
   505→          m.id === loadingMsg.id
   506→            ? {
   507→                ...m,
   508→                loading: false,
   509→                content: data.data.response,
   510→                meta: {
   511→                  strategy: data.data.strategy,
   512→                  aiModelsUsed: data.data.aiModelsUsed,
   513→                  webSources: data.data.webSources,
   514→                  creditsCost: data.data.creditsCost,
   515→                  latency: data.data.latency,
   516→                },
   517→              }
   518→            : m
   519→        )
   520→      )
   521→      if (data.data.creditsCost) {
   522→        onUpdateCredits(user.credits - data.data.creditsCost)
   523→      }
   524→    } catch {
   525→      setMessages((prev) =>
   526→        prev.map((m) =>
   527→          m.id === loadingMsg.id
   528→            ? { ...m, loading: false, content: '❌ Erreur de connexion au serveur.' }
   529→            : m
   530→        )
   531→      )
   532→    } finally {
   533→      setLoading(false)
   534→    }
   535→  }
   536→
   537→  const handleKeyDown = (e: KeyboardEvent<HTMLTextAreaElement>) => {
   538→    if (e.key === 'Enter' && !e.shiftKey) {
   539→      e.preventDefault()
   540→      sendMessage()
   541→    }
   542→  }
   543→
   544→  return (
   545→    <div className="flex flex-col h-full max-h-[calc(100vh-14rem)] md:max-h-[calc(100vh-12rem)]">
   546→      {/* Messages */}
   547→      <ScrollArea className="flex-1 factory-scroll" ref={scrollRef}>
   548→        <div className="p-4 space-y-4">
   549→          {messages.length === 0 && (
   550→            <motion.div {...slideUp} className="flex flex-col items-center justify-center py-20 text-center">
   551→              <div className="w-16 h-16 rounded-2xl bg-gradient-to-br from-emerald-500/20 to-violet-500/20 flex items-center justify-center mb-4">
   552→                <Bot className="w-8 h-8 text-emerald-400" />
   553→              </div>
   554→              <h3 className="text-lg font-semibold mb-2">Bienvenue sur FACTORY IA</h3>
   555→              <p className="text-muted-foreground text-sm max-w-md">
   556→                Posez votre question. Notre moteur multi-AI sélectionne automatiquement le meilleur modèle pour répondre.
   557→              </p>
   558→              <div className="flex flex-wrap gap-2 mt-4 justify-center">
   559→                {['Explique-moi React Server Components', 'Code un tri rapide en Python', 'Quelles sont les tendances IA 2025 ?'].map((s) => (
   560→                  <button
   561→                    key={s}
   562→                    onClick={() => setInput(s)}
   563→                    className="text-xs px-3 py-1.5 rounded-full border border-factory-border text-muted-foreground hover:text-foreground hover:border-emerald-500/50 transition-all"
   564→                  >
   565→                    {s}
   566→                  </button>
   567→                ))}
   568→              </div>
   569→            </motion.div>
   570→          )}
   571→
   572→          <AnimatePresence>
   573→            {messages.map((msg) => (
   574→              <motion.div
   575→                key={msg.id}
   576→                {...fadeIn}
   577→                className={`flex gap-3 ${msg.role === 'user' ? 'flex-row-reverse' : ''}`}
   578→              >
